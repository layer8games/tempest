122  void MakeOrthographic(T width, T height, T depth) {
123  T right = width / 2;
124  T left = -width / 2;
125  T top = height / 2;
126  T bottom = -height / 2;
127  T far = depth / 2;
128  T near = -depth / 2;
129 
130  //Reset Matrix
131  _m[0] = _m[1] = _m[2] = _m[3] = 0;
132 
133  _m[4] = _m[5] = _m[6] = _m[7] = 0;
134 
135  _m[8] = _m[9] = _m[10] = _m[11] = 0;
136 
137  _m[12] = _m[13] = _m[14] = _m[15] = 0;
138 
139  //Diagnal
140  _m[0] = 2 / (right - left);
141  _m[5] = 2 / (top - bottom);
142  _m[10] = 2 / (near - far);
143  _m[15] = 1;
144 
145  //Transform "Vector"
146  _m[12] = (left + right) / (left - right);
147  _m[13] = (bottom + top) / (bottom - top);
148  _m[14] = (near + far) / (far - near);
149 
150  }
